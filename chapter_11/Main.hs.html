1 module Main where 
    2 import QC
    3 import Test.QuickCheck
    4 
    5 anal :: Args 
    6 anal = Args {
    7     replay = Nothing,
    8     maxSuccess = 1000,
    9     maxDiscardRatio = 1,
   10     maxSize = 1000,
   11     chatty = True
   12 }
   13 
   14 minimal :: Args 
   15 minimal = Args {
   16     replay = Nothing,
   17     maxSuccess = 200,
   18     maxDiscardRatio = 1,
   19     maxSize = 200,
   20     chatty = True
   21 }
   22 
   23 runTest :: Args -> IO()
   24 runTest args = do 
   25         f prop_empty_id "empty_id ok?"
   26         f prop_char "char ok?"
   27         f prop_text "text ok?"
   28         f prop_line "line ok?"
   29         f prop_double "double ok?"
   30         f prop_hcat "hcat ok?"
   31         f prop_punctuate "punctuate ok?"
   32         where f prop str = do 
   33                 putStrLn str 
   34                 quickCheckWithResult args prop 
   35                 return ()
   36 main :: IO () 
   37 main = do 
   38      putStrLn "Choose test depth"
   39      putStrLn "1. Anal"
   40      putStrLn "2. Minimal"
   41      depth <- readLn
   42      if depth == 1 
   43      then runTest anal
   44      else runTest minimal
